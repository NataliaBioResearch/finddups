    $wixobj = Join-Path $base "FindDups.wixobj"
    $msi = Join-Path $base "FindDups.msi"

    [System.Windows.Forms.MessageBox]::Show("Compilant FindDups.wxs...")

    candle $wxs -out $wixobj

    if (Test-Path $msi) {

        [System.Windows.Forms.MessageBox]::Show("Error: no s'ha generat el MSI","Error")
